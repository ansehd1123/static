function showFormattedNames() { const htmlData = document.getElementById("htmlData").value; // HTML 파싱 const parser = new DOMParser(); const doc = parser.parseFromString(htmlData, 'text/html'); // name 속성이 존재하는 모든 태그 추출 const rawNames = Array.from(doc.querySelectorAll('[name]')) .map(el => el.getAttribute('name')); // 중복 제거 const uniqueNames = [...new Set(rawNames)]; // '[name]=testtest' 형식으로 변환 const formattedOutput = uniqueNames .map(name => `${name}=testtest`) .join('&'); document.getElementById("formattedResult").value = formattedOutput; }
